182 void GROManager::fetchStructure()
183 {
184  SystemStructureRecord *systemStructure;
185  COMStructureRecord *comStructure;
186 
187  unsigned short lastMoleculeType;
188  unsigned int particlesNumber;
189  QString lastResidueName ;
190  QString currentMoleculeName;
191  short moleculeIdentifier[ UINT_THREE ] = { -1, -1, -1 }; // to start by ZERO
192  unsigned int residueParticlesCounter[ UINT_THREE ] = { UINT_ONE, UINT_ONE, UINT_ONE };
193  bool isNewMolecule[ UINT_THREE ] = {false, false, false};
194  short residueCounter[ UINT_THREE ] = { SHORT_ZERO, SHORT_ZERO, SHORT_ZERO };
195  unsigned int residueNumber;
196  QString residueName;
197  unsigned int particleNumber;
198  QString particleName;
199  QString residueTag;
200 
201  unsigned int proteinsParticlesNumber(UINT_ZERO);
202  unsigned short proteinResidueID(UINT_ZERO );
203  unsigned short lipidResidueID(UINT_ZERO );
204 
205  unsigned int lipidsParticlesNumber(UINT_ZERO);
206  unsigned int proteinIndexCOM(UINT_ZERO);
207  unsigned int lipidIndexCOM( PROTEIN_INDEX_UPPER_BOUNDARY );
208  lastResidueName = EMPTY_STRING;
209  lastMoleculeType = MOLECULE_UNDEFINED;
210 
211  particlesNumber = ATOMS_NUMBER ;
212 
213  systemStructure = DataStructure::getSystemStructure();
214  comStructure = DataStructure::getCOMStructure();
215 
216  for( unsigned int atomIndex = UINT_ZERO; atomIndex < particlesNumber; atomIndex++)
217  {
218  const QString detail = getGROData().at( atomIndex );
219  unsigned short currentMoleculeType;
220  short currentMoleculeID;
221 
222  fetchAtomInfo( detail, residueNumber, residueName, particleNumber, particleName, residueTag );
223 
224  fetchCurrentMoleculeInfo( lastResidueName, residueName, lastMoleculeType, currentMoleculeType, currentMoleculeName,
225  currentMoleculeID, moleculeIdentifier, residueParticlesCounter, isNewMolecule, residueCounter );
226 
227 
228  //computing the correct Protein Residue ID and return the protein Index in the COM file.
229  computeProteinResidueID(currentMoleculeType, proteinsParticlesNumber, proteinResidueID , residueNumber);
230  //computing the correct Lipid residue ID and return the lipid index in the COM file.
231  computeLipidResidueID(currentMoleculeType, lipidsParticlesNumber, lipidResidueID);
232 
233  //Update lookup table
234  updateProteinResidueNameidentifierTable(currentMoleculeType, residueNumber, residueName );
235  //residue name here is either POPE or POPG and particle name is the real residue :)
236  updateLipidPOPEResidueNameidentifierTable(residueName, lipidResidueID, particleName );
237  updateLipidPOPGResidueNameidentifierTable(residueName, lipidResidueID, particleName );
238 
239  //Update system structure records
240  updateSystemStructureProteinRecord( systemStructure, comStructure, atomIndex, currentMoleculeID, proteinResidueID, currentMoleculeType, proteinsParticlesNumber, proteinIndexCOM );
241  updateSystemStructureLipidRecord( systemStructure, comStructure, atomIndex, currentMoleculeID, currentMoleculeType, lipidsParticlesNumber, lipidResidueID , lipidIndexCOM );
242 
243  lastMoleculeType = currentMoleculeType;
244  }
245 
246  getGROData().clear();
247 }
